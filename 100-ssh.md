SSH BatchMode:

```
tmpdir=${TMPDIR:-/tmp}/pssh.$$
mkdir -p $tmpdir
count=0
while IFS= read -r userhost; do
    ssh -n -o BatchMode=yes ${userhost} "$command" > ${tmpdir}/${userhost} 2>&1 &
    count=`expr $count + 1`
done < userhost.txt
while [ $count -gt 0]; do
    wait $pids
    count=`expr $count - 1`
done
echo "Result for hosts are in $tmpdir"
```
